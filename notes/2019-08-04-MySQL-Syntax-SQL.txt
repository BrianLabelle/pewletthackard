CREATE TABLE `pewlett_hackard`.`employees` (
  `emp_no` INT NOT NULL,
  `birth_date` DATETIME NOT NULL,
  `first_name` VARCHAR(45) NOT NULL,
  `last_name` VARCHAR(45) NOT NULL,
  `gender` VARCHAR(1) NOT NULL,
  `hire_date` DATETIME NOT NULL,
  PRIMARY KEY (`emp_no`));


CREATE TABLE `pewlett_hackard`.`departments` (
  `dept_no` VARCHAR(4) NOT NULL,
  `dept_name` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`dept_no`));



CREATE TABLE `pewlett_hackard`.`dept_emp` (
  `emp_no` INT NOT NULL,
  `dept_no` VARCHAR(4) NOT NULL,
  `from_date` DATETIME NOT NULL,
  `to_date` DATETIME NOT NULL DEFAULT '9999-01-01');

CREATE TABLE `pewlett_hackard`.`dept_manager` (
  `dept_no` VARCHAR(4) NOT NULL,
  `emp_no` INT NULL,
  `from_date` DATETIME NULL,
  `to_date` DATETIME NOT NULL DEFAULT '9999-01-01');

CREATE TABLE `pewlett_hackard`.`salaries` (
  `emp_no` INT NOT NULL,
  `salary` DECIMAL(2) NOT NULL,
  `from_date` DATETIME NOT NULL,
  `to_date` DATETIME NOT NULL DEFAULT '9999-01-01');


CREATE TABLE `pewlett_hackard`.`titles` (
  `emp_no` INT NOT NULL,
  `title` VARCHAR(45) NOT NULL,
  `from_date` DATETIME NOT NULL,
  `to_date` DATETIME NOT NULL DEFAULT '9999-01-01');
